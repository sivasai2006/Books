object MyClass {
def main(args: Array[String]): Unit = {
        val in = scala.io.StdIn
        val n1 = in.readInt()
        for (_ <- 0 until n1) {
          var num = scala.collection.mutable.Set[Int]()
          val n2 = in.readInt()
          for(_ <- 0 until n2) {
            var arr= (in.readLine().split(" ").map(_.toInt))
            for (i <- arr(0) to arr(1)) {num += i}
          }
          println(num.size)
        }
    }
}
